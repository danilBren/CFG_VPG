grammar ltl_tagged.txt;


file_
	: proposition EOF
	;

proposition
	: <'(' proposition ')'> proposition`
	| _new_rule_2 proposition proposition`
	| 'false' proposition`
	| LTL_NOT proposition proposition`
	| 'true' proposition`
	| ATOMIC proposition`
	;

_new_rule_2
	: LTL_GLOBALLY
	| LTL_NEXT
	| LTL_FINALLY
	;

proposition`
	:
	| _new_rule_3 proposition proposition`
	| _new_rule_1 proposition proposition`
	;

_new_rule_1
	: LTL_RIGHTWARDS_SINGLE_ARROW
	| LTL_AND
	| LTL_OR
	;

_new_rule_3
	: LTL_WEAK
	| LTL_UNTIL
	| LTL_RELEASE
	;
