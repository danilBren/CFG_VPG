	: expression _new_rule_1* EOF
	;

	: SEMI expression
	;

expression

	| _new_rule_3* atom expression`
	;

atom
	: variable
	| scientific
	;

scientific
	: SCIENTIFIC_NUMBER
	;

variable

	;

_new_rule_3
	: PLUS
	| MINUS
	;

expression`
	:
	| _new_rule_3 expression expression`
	| POW expression expression`
	| _new_rule_2 expression expression`
	;

_new_rule_2
	: DIV
	| TIMES
	;
