grammar bnf_tagged.txt;


rulelist
	: rule_* EOF
	;

rule_
	: lhs ASSIGN rhs
	;

rhs
	: alternatives
	;

alternatives
	: alternative _new_rule_1*
	;

_new_rule_1
	: '|' alternative
	;

alternative
	: element*
	;

element
	: zeroormore
	| oneormore
	| optional_
	| id_
	| text_
	;

text_
	: ID
	;

id_
	: <'<' ruleid '>'>
	;

ruleid
	: ID
	;

optional_
	: <'[' alternatives ']'>
	;

oneormore
	: <'(' alternatives ')'>
	;

zeroormore
	: <'{' alternatives '}'>
	;

lhs
	: id_
	;
